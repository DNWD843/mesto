Name Type Description
baseURL String базовый URL сервера
cardsURL String URL для получения/добавления карточек
userURL String URLдля получения/добавления данных пользователя

Name Type Description

Example
const api = new Api({
URLs: {
  baseURL: 'https://mesto...',
  cardsURL: 'https://mesto.../cards/',
  userURL: 'https://mesto.../users/me',
  likesURL: 'https://mesto.../cards/likes/',
  avatarURL: 'https://mesto.../users/me/avatar'
},
headers: {
  "authorization": '<Ваш код авторизации>'
},
token: '<Ваш токен>'
});
